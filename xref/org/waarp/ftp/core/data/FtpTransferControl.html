162 	public void setOpenedDataChannel(Channel channel,
163 			DataNetworkHandler dataNetworkHandler) {
164 		logger.debug("SetOpenedDataChannel: "+(channel != null ? channel.getRemoteAddress() : "no channel"));
165 		if (channel != null) {
166 			session.getDataConn().setDataNetworkHandler(dataNetworkHandler);
167 			waitForOpenedDataChannel.setChannel(channel);

174 	/**
175 	 * Wait that the new opened connection is ready (same method in {@link FtpDataAsyncConn} from
176 	 * openConnection)
177 	 * 
178 	 * @return the new opened Channel
179 	 * @throws InterruptedException
180 	 */
181 	public Channel waitForOpenedDataChannel() throws InterruptedException {
182 		Channel channel = null;
183 		if (waitForOpenedDataChannel.await(
184 				session.getConfiguration().TIMEOUTCON + 1000,
185 				TimeUnit.MILLISECONDS)) {
186 			if (waitForOpenedDataChannel.isSuccess()) {
187 				channel = waitForOpenedDataChannel.getChannel();
188 			} else {
189 				logger.warn("data connection is in error");
190 			}
191 		} else {
192 			logger.warn("Timeout occurs during data connection");
193 		}
194 		waitForOpenedDataChannel = new WaarpChannelFuture(true);
195 		return channel;
196 	}

214 	/**
215 	 * Wait for the client to be connected (Passive) or Wait for the server to be connected to the
216 	 * client (Active)
217 	 * 
218 	 * @return True if the connection is OK
219 	 * @throws Reply425Exception
220 	 */
221 	public boolean openDataConnection() throws Reply425Exception {
222 		// Prepare this Data channel to be closed ;-)
223 		// In fact, prepare the future close op which should occur since it is
224 		// now opened
225 		closedDataChannel = new WaarpFuture(true);
226 		FtpDataAsyncConn dataAsyncConn = session.getDataConn();
227 		if (!dataAsyncConn.isStreamFile()) {
228 			// FIXME isConnected or isDNHReady ?
229 			if (dataAsyncConn.isConnected()) {
230 				// Already connected
231 				// logger.debug("Connection already open");
232 				session.setReplyCode(
233 						ReplyCode.REPLY_125_DATA_CONNECTION_ALREADY_OPEN,
234 						dataAsyncConn.getType().name() +
235 								" mode data connection already open");
236 				return true;
237 			}
238 		} else {
239 			// Stream, Data Connection should not be opened
240 			if (dataAsyncConn.isConnected()) {
241 				logger
242 						.error("Connection already open but should not since in Stream mode");
243 				setTransferAbortedFromInternal(false);
244 				throw new Reply425Exception(
245 						"Connection already open but should not since in Stream mode");
246 			}
247 		}
248 		// Need to open connection
249 		session.setReplyCode(ReplyCode.REPLY_150_FILE_STATUS_OKAY, "Opening " +
250 				dataAsyncConn.getType().name() + " mode data connection");
251 		if (dataAsyncConn.isPassiveMode()) {
252 			if (!dataAsyncConn.isBind()) {
253 				// No passive connection prepared
254 				throw new Reply425Exception(
255 						"No passive data connection prepared");
256 			}
257 			// Wait for the connection to be done by the client
258 			// logger.debug("Passive mode standby");
259 			try {
260 				dataChannel = waitForOpenedDataChannel();
261 				dataAsyncConn.setNewOpenedDataChannel(dataChannel);
262 			} catch (InterruptedException e) {
263 				logger.warn("Connection abort in passive mode", e);
264 				// Cannot open connection
265 				throw new Reply425Exception(
266 						"Cannot open passive data connection");
267 			}
268 			// logger.debug("Passive mode connected");
269 		} else {
270 			// Wait for the server to be connected to the client
271 			InetAddress inetAddress = dataAsyncConn.getLocalAddress()
272 					.getAddress();
273 			InetSocketAddress inetSocketAddress = dataAsyncConn
274 					.getRemoteAddress();
275 			if (session.getConfiguration().getFtpInternalConfiguration()
276 					.hasFtpSession(inetAddress, inetSocketAddress)) {
277 				throw new Reply425Exception(
278 						"Cannot open active data connection since remote address is already in use: "
279 								+
280 								inetSocketAddress);
281 			}
282 			// logger.debug("Active mode standby");
283 			ClientBootstrap clientBootstrap = session.getConfiguration()
284 					.getFtpInternalConfiguration().getActiveBootstrap(session.isDataSsl());
285 			session.getConfiguration().setNewFtpSession(inetAddress,
286 					inetSocketAddress, session);
287 			// Set the session for the future dataChannel
288 			String mylog = session.toString();
289 			logger.debug("DataConn for: " + session.getCurrentCommand().getCommand() + " to "
290 					+ inetSocketAddress.toString());
291 			ChannelFuture future = clientBootstrap.connect(inetSocketAddress,
292 					dataAsyncConn.getLocalAddress());
293 			try {
294 				future.await();
295 			} catch (InterruptedException e1) {
296 			}
297 			if (!future.isSuccess()) {
298 				logger.warn("Connection abort in active mode from future while session: " +
299 						session.toString() +
300 						"\nTrying connect to: " + inetSocketAddress.toString() +
301 						"\nWas: " + mylog,
302 						future.getCause());
303 				// Cannot open connection
304 				session.getConfiguration().delFtpSession(inetAddress,
305 						inetSocketAddress);
306 				throw new Reply425Exception(
307 						"Cannot open active data connection");
308 			}
309 			try {
310 				dataChannel = waitForOpenedDataChannel();
311 				dataAsyncConn.setNewOpenedDataChannel(dataChannel);
312 			} catch (InterruptedException e) {
313 				logger.warn("Connection abort in active mode", e);
314 				// Cannot open connection
315 				session.getConfiguration().delFtpSession(inetAddress,
316 						inetSocketAddress);
317 				throw new Reply425Exception(
318 						"Cannot open active data connection");
319 			}
320 			// logger.debug("Active mode connected");
321 		}
322 		if (dataChannel == null) {
323 			// Cannot have a new Data connection since shutdown
324 			if (!dataAsyncConn.isPassiveMode()) {
325 				session.getConfiguration().getFtpInternalConfiguration()
326 						.delFtpSession(
327 								dataAsyncConn.getLocalAddress().getAddress(),
328 								dataAsyncConn.getRemoteAddress());
329 			}
330 			throw new Reply425Exception(
331 					"Cannot open data connection, shuting down");
332 		}
333 		return true;
334 	}

337 	/**
338 	 * Run the command from an executor
339 	 */
340 	private void runExecutor() {
341 		// Unlock Mode Codec
342 		try {
343 			session.getDataConn().getDataNetworkHandler().unlockModeCodec();
344 		} catch (FtpNoConnectionException e) {
345 			setTransferAbortedFromInternal(false);
346 			return;
347 		}
348 		// Run the command
349 		if (executorService == null) {
350 			executorService = Executors.newSingleThreadExecutor();
351 		}
352 		endOfCommand = new WaarpFuture(true);
353 		final ArrayList<Callable<Object>> tasks = new ArrayList<Callable<Object>>(1);
354 		tasks.add(Executors.callable(new FtpTransferExecutor(session,
355 				executingCommand)));
356 		try {
357 			executorService.invokeAll(tasks);
358 		} catch (InterruptedException e1) {
359 		}
360 		// XXX TRY FIX TO IMPROVE
361 		/*
362 		executorService.execute(new FtpTransferExecutor(session,
363 			executingCommand));
364 			*/
365 		try {
366 			commandFinishing.await();
367 		} catch (InterruptedException e) {
368 		}
369 	}

462 	/**
463 	 * Called when a transfer is finished from setEndOfTransfer
464 	 * 
465 	 * @return True if it was already called before
466 	 * @throws FtpNoTransferException
467 	 */
468 	private boolean checkFtpTransferStatus() throws FtpNoTransferException {
469 		if (isCheckAlreadyCalled) {
470 			logger.warn("Check: ALREADY CALLED");
471 			return true;
472 		}
473 		if (isExecutingCommandFinished) {
474 			// already done
475 			logger.warn("Check: already Finished");
476 			if (commandFinishing != null) {
477 				commandFinishing.cancel();
478 			}
479 			throw new FtpNoTransferException("No transfer running");
480 		}
481 		if (!isDataNetworkHandlerReady) {
482 			// already done
483 			logger.warn("Check: already DNH not ready");
484 			throw new FtpNoTransferException("No connection");
485 		}
486 		isCheckAlreadyCalled = true;
487 		FtpTransfer executedTransfer = getExecutingFtpTransfer();
488 		// logger.debug("Check: command {}", executedTransfer.getCommand());
489 		// DNH is ready and Transfer is running
490 		if (FtpCommandCode.isListLikeCommand(executedTransfer.getCommand())) {
491 			if (executedTransfer.getStatus()) {
492 				// Special status for List Like command
493 				// logger.debug("Check: List OK");
494 				closeTransfer();
495 				return false;
496 			}
497 			// logger.debug("Check: List Ko");
498 			abortTransfer();
499 			return false;
500 		} else if (FtpCommandCode.isRetrLikeCommand(executedTransfer
501 				.getCommand())) {
502 			FtpFile file = null;
503 			try {
504 				file = executedTransfer.getFtpFile();
505 			} catch (FtpNoFileException e) {
506 				// logger.debug("Check: Retr no FtpFile for Retr");
507 				abortTransfer();
508 				return false;
509 			}
510 			try {
511 				if (file.isInReading()) {
512 					logger
513 							.debug("Check: Retr FtpFile still in reading KO");
514 					abortTransfer();
515 				} else {
516 					logger
517 							.debug("Check: Retr FtpFile no more in reading OK");
518 					closeTransfer();
519 				}
520 			} catch (CommandAbstractException e) {
521 				logger.warn("Retr Test is in Reading problem", e);
522 				closeTransfer();
523 			}
524 			return false;
525 		} else if (FtpCommandCode.isStoreLikeCommand(executedTransfer
526 				.getCommand())) {
527 			// logger.debug("Check: Store OK");
528 			closeTransfer();
529 			return false;
530 		} else {
531 			logger.warn("Check: Unknown command");
532 			abortTransfer();
533 		}
534 		return false;
535 	}

537 	/**
538 	 * Abort the current transfer
539 	 */
540 	private void abortTransfer() {
541 		// logger.debug("Will abort transfer and write: ", write);
542 		FtpFile file = null;
543 		FtpTransfer current = null;
544 		try {
545 			current = getExecutingFtpTransfer();
546 			file = current.getFtpFile();
547 			file.abortFile();
548 		} catch (FtpNoTransferException e) {
549 			logger.warn("Abort problem", e);
550 		} catch (FtpNoFileException e) {
551 		} catch (CommandAbstractException e) {
552 			logger.warn("Abort problem", e);
553 		}
554 		if (current != null) {
555 			current.setStatus(false);
556 		}
557 		endDataConnection();
558 		session.setReplyCode(
559 				ReplyCode.REPLY_426_CONNECTION_CLOSED_TRANSFER_ABORTED,
560 				"Transfer aborted for " +
561 						(current == null ? "Unknown command" : current
562 								.toString()));
563 		if (current != null) {
564 			if (!FtpCommandCode.isListLikeCommand(current.getCommand())) {
565 				try {
566 					session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
567 				} catch (CommandAbstractException e) {
568 					session.setReplyCode(e);
569 				}
570 			}
571 		}
572 		finalizeExecution();
573 	}

575 	/**
576 	 * Finish correctly a transfer
577 	 * 
578 	 */
579 	private void closeTransfer() {
580 		// logger.debug("Will close transfer and write: {}", write);
581 		FtpFile file = null;
582 		FtpTransfer current = null;
583 		try {
584 			current = getExecutingFtpTransfer();
585 			file = current.getFtpFile();
586 			file.closeFile();
587 		} catch (FtpNoTransferException e) {
588 			logger.warn("Close problem", e);
589 		} catch (FtpNoFileException e) {
590 		} catch (CommandAbstractException e) {
591 			logger.warn("Close problem", e);
592 		}
593 		if (current != null) {
594 			current.setStatus(true);
595 		}
596 		if (session.getDataConn().isStreamFile()) {
597 			endDataConnection();
598 		}
599 		session.setReplyCode(ReplyCode.REPLY_226_CLOSING_DATA_CONNECTION,
600 				"Transfer complete for " +
601 						(current == null ? "Unknown command" : current
602 								.toString()));
603 		if (current != null) {
604 			if (!FtpCommandCode.isListLikeCommand(current.getCommand())) {
605 				try {
606 					session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
607 				} catch (CommandAbstractException e) {
608 					session.setReplyCode(e);
609 				}
610 			} else {
611 				// Special wait to prevent fast LIST following by STOR or RETR command
612 				try {
613 					Thread.sleep(FtpInternalConfiguration.RETRYINMS);
614 				} catch (InterruptedException e) {
615 				}
616 			}
617 		}
618 		finalizeExecution();
619 	}

636 	 * 
637 	 * @param write
638 	 *            True means the message is write back to the control command, false it is only
639 	 *            prepared
640 	 */
641 	public void setTransferAbortedFromInternal(boolean write) {
642 		// logger.debug("Set transfer aborted internal {}", write);
643 		abortTransfer();
644 		if (write) {
645 			session.getNetworkHandler().writeIntermediateAnswer();
646 		}
647 		if (endOfCommand != null) {
648 			endOfCommand.cancel();
649 		}

664 	 *

667 	public void waitForEndOfTransfer() throws InterruptedException {

675 	}

679 	 * Finalize execution
680 	 *

682 	private void finalizeExecution() {
