42  /**
43   * @author "Frederic Bregier"
44   *
45   */
46  public class FtpsPipelineFactory implements ChannelPipelineFactory {
47  	/**
48  	 * CRLF, CRNUL, LF delimiters
49  	 */
50  	private static final ChannelBuffer[] delimiter = new ChannelBuffer[] {
51  			ChannelBuffers.wrappedBuffer(ReplyCode.CRLF.getBytes()),
52  			ChannelBuffers.wrappedBuffer(ReplyCode.CRNUL.getBytes()),
53  			ChannelBuffers.wrappedBuffer(ReplyCode.LF.getBytes()) };
54  
55  	private static final FtpControlStringDecoder ftpControlStringDecoder = new FtpControlStringDecoder();
56  
57  	private static final FtpControlStringEncoder ftpControlStringEncoder = new FtpControlStringEncoder();
58  
59  	public static WaarpSslContextFactory waarpSslContextFactory;
60  	public static WaarpSecureKeyStore WaarpSecureKeyStore;
61  	private final ExecutorService executorService;
62  
63  	/**
64  	 * Business Handler Class if any (Target Mode only)
65  	 */
66  	private final Class<? extends BusinessHandler> businessHandler;
67  
68  	/**
69  	 * Configuration
70  	 */
71  	private final FtpConfiguration configuration;
72  
73  	/**
74  	 * Constructor which Initializes some data for Server only
75  	 * 
76  	 * @param businessHandler
77  	 * @param configuration
78  	 * @param executor
79  	 */
80  	public FtpsPipelineFactory(Class<? extends BusinessHandler> businessHandler,
81  			FtpConfiguration configuration, ExecutorService executor) {
82  		this.businessHandler = businessHandler;
83  		this.configuration = configuration;
84  		this.executorService = executor;
85  	}
86  
87  	/**
88  	 * Create the pipeline with Handler, ObjectDecoder, ObjectEncoder.
89  	 * 
90  	 * @see org.jboss.netty.channel.ChannelPipelineFactory#getPipeline()
91  	 */
92  	public ChannelPipeline getPipeline() throws Exception {
93  		ChannelPipeline pipeline = Channels.pipeline();
94  		// Server: no renegotiation still, but possible clientAuthent
95  		SslHandler handler = waarpSslContextFactory.initPipelineFactory(true,
96  				waarpSslContextFactory.needClientAuthentication(),
97  				false, executorService);
98  		// NO since we need to inform through SNMP: handler.setIssueHandshake(true);
99  		pipeline.addLast("ssl", handler);
100 
101 		// Add the text line codec combination first,
102 		pipeline.addLast("framer", new DelimiterBasedFrameDecoder(8192,
103 				delimiter));
104 		pipeline.addLast("decoder", ftpControlStringDecoder);
105 		pipeline.addLast("encoder", ftpControlStringEncoder);
106 		// Threaded execution for business logic
107 		pipeline.addLast("pipelineExecutor", new ExecutionHandler(configuration
108 				.getFtpInternalConfiguration().getPipelineExecutor()));
109 		// and then business logic. New one on every connection
110 		BusinessHandler newbusiness = businessHandler.newInstance();
111 		SslNetworkHandler newNetworkHandler = new SslNetworkHandler(new FtpSession(
112 				configuration, newbusiness));
113 		pipeline.addLast("handler", newNetworkHandler);
114 		return pipeline;
115 	}
